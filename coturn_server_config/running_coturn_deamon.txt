turnserver -v

with configuration file which is in the following path: /etc/turnserver.conf

changes requred:
#internal ip is the listening ip:
listening-ip=172.31.44.89
external-ip=54.93.214.159

#internal ip here also
relay-ip=172.31.44.89

#for ip under NAT like amazon: internal ip/ external ip
external-ip=54.93.214.159/172.31.44.89

#using static authentication
lt-cred-mech

#define some realm
realm=realm

#define user and password (usr:pass)
user=user:limor1
